Are there differences in activity patterns between weekdays and weekends?

  • Solution makes use of “timeDate” R package (containing isWeekend/isWeekday functions)
my_act$wknd <- isWeekend(my_act$date)
my_act$wknd <- gsub("FALSE", "Weekday", my_act$wknd)
my_act$wknd <- gsub("TRUE" , "Weekend", my_act$wknd)


par(col="blue", mfrow=c(2,1))
daytypes <- c("Weekend", "Weekday")
for (i in daytypes) {
    steps.i <- aggregate(steps ~ interval,
                           data=my_act,
                           subset=my_act$wknd==i,
                           FUN=mean)
    plot(steps.i, type="l", main=i)
}


my_act <- aggregate(
    data=my_act,
    steps ~ wknd + interval,
    FUN=mean
)
